<task type="auto">
  <name>Task 3: GitHub Actions CI pipeline</name>
  <files>.github/workflows/deploy.yml</files>
  <action>
    Create .github/workflows/deploy.yml:
    ```yaml
    name: CI

    on:
      push:
        branches: [main]
      pull_request:
        branches: [main]

    jobs:
      build-backend:
        runs-on: ubuntu-latest
        steps:
          - uses: actions/checkout@v4
          - uses: actions/setup-java@v4
            with:
              java-version: '21'
              distribution: 'temurin'
              cache: maven
          - name: Build backend
            run: cd backend && ./mvnw clean package -DskipTests

      build-frontend:
        runs-on: ubuntu-latest
        steps:
          - uses: actions/checkout@v4
          - uses: actions/setup-node@v4
            with:
              node-version: '20'
              cache: 'npm'
              cache-dependency-path: frontend/package-lock.json
          - name: Install dependencies
            run: cd frontend && npm ci
          - name: Build frontend
            run: cd frontend && npm run build
    ```
  </action>
  <verify>cat .github/workflows/deploy.yml</verify>
  <acceptance_criteria>
    - CI workflow builds both backend and frontend on push to main
  </acceptance_criteria>
  <done>GitHub Actions CI pipeline created</done>
</task>
